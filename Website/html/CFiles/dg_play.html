dg_play

This function create a new window.
This function require the functions dg_init and dg_loop, otherwise, you won't be able to compile

Usage

  1. void dg_play(unsigned int width, unsigned int height, char *name)
  • width : width of the window, in pixels
  • height : height of the window, in pixels
  • name : Name of the window (will be displayed at the top of the window)

Exemple

  1. dg_play(1920, 1980, "window");

Source code

  1. void dg_play(unsigned int width, unsigned int height, char *name)
  2. {
  3. sfVideoMode mode = {width, height, 32};
  4. sfRenderWindow *window = sfRenderWindow_create(mode, name,
  5. sfResize | sfClose, NULL);
  6. framebuffer_t *back_fb = framebuffer_create(mode.width, mode.height);
  7. int (*init)(framebuffer_t *) = &dg_init;
  8. int (*loop)(framebuffer_t *) = &dg_loop;
  9. sfEvent event;
  11. (*init)(back_fb);
  12. sfRenderWindow_clear(window, sfBlack);
  13. while (sfRenderWindow_isOpen(window)) {
  14. while (sfRenderWindow_pollEvent(window, &event)) {
  15. if (event.type == sfEvtClosed)
  16. sfRenderWindow_close(window);
  17. }
  18. (*loop)(back_fb);
  19. framebuffer_update(back_fb, window);
  20. sfRenderWindow_display(window);
  21. }
  22. framebuffer_destroy(back_fb);
  23. sfRenderWindow_destroy(window);
  24. }
